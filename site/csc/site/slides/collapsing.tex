\section{Collapsing Superstring Conjecture}

\begin{frame}[label=hga]{Collapsing Algorithm}
\begin{itemize}
\item Take any solution
\item Double it
\item Collapse everything: replace every pair of edges $aS \to aS b \to Sb$ by a~pair of edges $aS \to S \to S b$ \alert{if this preserves connectivity}
\end{itemize}
\end{frame}

\begin{frame}{Collapsing Superstring Conjecture}
\begin{itemize}
\item \alert{Conjecture:} the result of the double-and-collapse procedure is the same for all initial solutions
\item If the conjecture holds, then the Collapsing Algorithm is 2-approximate
\end{itemize}
\end{frame}
